% Header and footer for when a page split occurs within the issue environment
\newcommand{\enterIssueHeader}[1]{
\nobreak\extramarks{#1}{\ldots}\nobreak
\nobreak\extramarks{#1 (continued)}{}\nobreak
}

% Header and footer for when a page split occurs between issue environments
\newcommand{\exitIssueHeader}[1]{
\nobreak\extramarks{#1 (continued)}{#1 continued on next page\ldots}\nobreak
\nobreak\extramarks{#1}{}\nobreak
}

\setcounter{secnumdepth}{3} % Removes default section numbers
\newcounter{issueCounter} % Creates a counter to keep track of the number of issues

\newcommand{\issueName}{}
\newenvironment{issue}[1][Issue \arabic{issueCounter}]{ % Makes a new environment called issue which takes 1 argument (custom name) but the default is "Issue #"
\stepcounter{issueCounter} % Increase counter for number of issues
\renewcommand{\issueName}{#1} % Assign \issueName the name of the issue
\section*{\issueName} % Make a section in the document with the custom issue count
\enterIssueHeader{\issueName} % Header and footer within the environment
}{
\exitIssueHeader{\issueName} % Header and footer after the environment
}

\newcommand{\problemAnswer}[1]{ % Defines the answer to a problem with the content as the only argument
\noindent\framebox[\columnwidth][c]{\begin{minipage}{0.93\columnwidth}#1\end{minipage}} % Makes the box around the problem answer and puts the content inside
}

\newcommand{\issueSectionName}{}
\newenvironment{issueSection}[1]{ % New environment for sections within issues, takes 1 argument - the name of the section
\renewcommand{\issueSectionName}{#1} % Assign \homeworkSectionName to the name of the section from the environment argument
\subsection{\issueSectionName} % Make a subsection with the custom name of the subsection
\enterIssueHeader{\issueName\ [\issueSectionName]} % Header and footer within the environment
}{
\enterIssueHeader{\issueName} % Header and footer after the environment
}
